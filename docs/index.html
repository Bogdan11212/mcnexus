mcnexus

Production-ready Python library for Minecraft server management.

📦 Installation

You can install mcnexus directly from GitHub until it is published on PyPI:

pip install git+https://github.com/Bogdan11212/mcnexus.git

⚡ RCON Module

A robust, asynchronous RCON implementation with multi-packet support and auto-reconnect.

Basic Usage

import asyncio
from mcnexus import RCONClient

async def main():
    # Use context manager for automatic connection/disconnection
    async with RCONClient("127.0.0.1", 25575, "your_password") as client:
        response = await client.command("list")
        print(f"Clean Output: {response.clean}")
        print(f"Raw Output: {response.raw}")
        print(f"Execution Time: {response.time_taken:.4f}s")

asyncio.run(main())

Server Pool (Manage multiple servers)

from mcnexus import RCONPool

async with RCONPool() as pool:
    pool.add_server("Lobby", "10.0.0.1", 25575, "pass1")
    pool.add_server("Survival", "10.0.0.2", 25575, "pass2")
    
    # Broadcast command to all servers
    results = await pool.broadcast("say Global Announcement!")
    for name, res in results.items():
        print(f"{name} responded: {res}")

Watcher (Simulate Events)

from mcnexus import RCONWatcher, RCONClient

async def on_player_join(response):
    print(f"Detected activity: {response.clean}")

async with RCONClient(...) as client:
    watcher = RCONWatcher(client, interval=2.0)
    # Triggers if 'joined' appears in 'list' command output
    watcher.on_output_contains("list", "joined", on_player_join)
    await watcher.start()

🌍 Status Module

Retrieve server info using Modern (1.7+) or Legacy (pre-1.7) SLP protocols with SRV support.

import asyncio
from mcnexus import status, status_bulk

async def check_server():
    # Automatically handles SRV records (e.g. hypixel.net -> IP:Port)
    res = await status("mc.hypixel.net")
    
    if res.online:
        print(f"Players: {res.players_online}/{res.players_max}")
        print(f"Version: {res.version_name}")
        print(f"Ping: {res.ping:.2f}ms")
        print(f"Clean MOTD: {res.motd_clean}")

asyncio.run(check_server())

Bulk Status Check

# Check hundreds of servers concurrently
servers = [("mc.hypixel.net", 25565), ("mineplex.com", 25565)]
results = await status_bulk(servers)

🛠️ Utilities

Useful helper functions included in the library.

from mcnexus import strip_colors

# Remove Minecraft formatting codes (e.g. §a)
clean_text = strip_colors("§aGreen §lBold §rText")
print(clean_text) # Output: Green Bold Text
